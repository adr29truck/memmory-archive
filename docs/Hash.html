def objectify(clazz)

  if Array.class_exists?(clazz)
    clazzer = Object.const_get(clazz)
  else
    clazzer = ClassFactory.create_class clazz.downcase.capitalize, ApplicationController
  end

end

def transform_keys
  result = {}
  each_key do |key|
    result[yield(key)] = self[key]
  end
  result
end
